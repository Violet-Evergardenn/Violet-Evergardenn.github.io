DNA是生物遗传信息的载体，几十年来一直被用于刑事司法。但是，DNA分析究竟是如何工作的呢？给定一个DNA序列，法医调查人员如何确定它属于谁？

在名为 dna 的文件夹中调用 dna.py 的文件中，实现一个程序，用于识别 DNA 序列所属的人。

给一串dna找到他对应的关键基因所对应的数量相同的人

import csv
import sys


def main():

    # TODO: Check for command-line usage
    num = len(sys.argv)
    if(num != 3):
        print("please use python dna.py x.csv y.txt")
        exit(0)
    # TODO: Read database file into a variable

    rows = [] # 记录行里面的内容
    header = [] # 记录头
    num_len = 0 # 记录有多少行
    with open(sys.argv[1],"r") as file:
        reader = csv.DictReader(file)
        for row in reader:
            num_len += 1
            rows.append(row)
            
    with open(sys.argv[1],"r") as file:
        reader = csv.reader(file)
        header.append(next(reader))
    # TODO: Read DNA sequence file into a variable

    with open(sys.argv[2],"r") as file:
        wok = file.read()

    # TODO: Find longest match of each STR in DNA sequence

    num = len(header[0])
    check = []
    for i in range(1,num):
        wooo = longest_match(wok,header[0][i])
        check.append(wooo)

    # TODO: Check database for matching profiles
    ans = 0
    for i in range(num_len):
        so = True
        for j in range(1,num):
            if(int(rows[i][header[0][j]]) != int(check[j-1])):
                so = False
                break
        if(so == True):
            print(rows[i]["name"])
            ans += 1
    if(ans == 0):
        print("No match")
    return


def longest_match(sequence, subsequence):
    """Returns length of longest run of subsequence in sequence."""
    #处理重要的基因连续的数量
    # Initialize variables
    longest_run = 0
    subsequence_length = len(subsequence)
    sequence_length = len(sequence)

    # Check each character in sequence for most consecutive runs of subsequence
    for i in range(sequence_length):

        # Initialize count of consecutive runs
        count = 0

        # Check for a subsequence match in a "substring" (a subset of characters) within sequence
        # If a match, move substring to next potential match in sequence
        # Continue moving substring and checking for matches until out of consecutive matches
        while True:

            # Adjust substring start and end
            start = i + count * subsequence_length
            end = start + subsequence_length

            # If there is a match in the substring
            if sequence[start:end] == subsequence:
                count += 1

            # If there is no match in the substring
            else:
                break

        # Update most consecutive matches found
        longest_run = max(longest_run, count)

    # After checking for runs at each character in seqeuence, return longest run found
    return longest_run


main()
